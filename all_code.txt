import React, { useEffect, useState, useRef, Suspense, lazy } from 'react';
import Lenis from 'lenis';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { Preloader } from './components/Preloader';
import { Hero } from './components/Hero';
import { Services } from './components/Services';
import { AnimatedStats } from './components/AnimatedStats';
import { BentoGrid } from './components/BentoGrid';
import { LighthouseWidget } from './components/LighthouseWidget';
import { TechStack } from './components/TechStack';

import { Contact } from './components/Contact';
import { StarField } from './components/StarField';
import { AmbientAudio } from './components/AmbientAudio';

// Lazy load heavy 3D components
const Process = lazy(() => import('./components/Process').then(m => ({ default: m.Process })));
const GlobeSection = lazy(() => import('./components/GlobeSection').then(m => ({ default: m.GlobeSection })));

gsap.registerPlugin(ScrollTrigger);

function SectionDivider() {
  return <div className="section-divider" />;
}

export default function App() {
  const [isLoaded, setIsLoaded] = useState(false);
  const [scrolled, setScrolled] = useState(false);

  useEffect(() => {
    const lenis = new Lenis({
      duration: 1.8,
      easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
      orientation: 'vertical',
      gestureOrientation: 'vertical',
      smoothWheel: true,
      wheelMultiplier: 0.8,
      touchMultiplier: 2,
      infinite: false,
    });

    function raf(time: number) {
      lenis.raf(time);
      requestAnimationFrame(raf);
    }

    requestAnimationFrame(raf);

    return () => { lenis.destroy(); };
  }, []);

  // Track scroll for nav glass effect
  useEffect(() => {
    const onScroll = () => setScrolled(window.scrollY > 80);
    window.addEventListener('scroll', onScroll, { passive: true });
    return () => window.removeEventListener('scroll', onScroll);
  }, []);

  // GSAP scroll-triggered reveal system
  useEffect(() => {
    if (!isLoaded) return;

    const revealElements = document.querySelectorAll('[data-reveal]');
    
    revealElements.forEach((el) => {
      ScrollTrigger.create({
        trigger: el,
        start: 'top 85%',
        onEnter: () => el.classList.add('revealed'),
      });
    });

    return () => ScrollTrigger.getAll().forEach(t => t.kill());
  }, [isLoaded]);

  const scrollTo = (id: string) => {
    const el = document.getElementById(id);
    el?.scrollIntoView({ behavior: 'smooth' });
  };

  const scrollToTop = () => {
    window.scrollTo({ top: 0, behavior: 'smooth' });
  };

  return (
    <main className="relative min-h-screen bg-void-black text-ice-white font-sans selection:bg-electric-blue/30">
      
      
      {!isLoaded && <Preloader onComplete={() => setIsLoaded(true)} />}

      {/* Global Star Field */}
      <StarField />

      {/* Navigation â€” Premium Glassmorphism on Scroll */}
      <nav className={`fixed top-0 left-0 w-full z-40 transition-all duration-700 ${
        scrolled 
          ? 'nav-glass py-4 px-6 md:px-8' 
          : 'p-6 md:p-8'
      }`}>
        <div className="max-w-7xl mx-auto flex justify-between items-center">
          <button 
            className="text-2xl font-bold tracking-[0.2em] uppercase text-ice-white pointer-events-auto group flex items-center gap-3 relative"
            onClick={scrollToTop}
          >
            <span className="relative z-10">
              TECHY
              <span className="absolute -bottom-1 left-0 w-0 h-[2px] bg-gradient-to-r from-electric-blue to-cyber-purple group-hover:w-full transition-all duration-500" />
            </span>
            <div className="absolute inset-0 bg-electric-blue/20 blur-xl rounded-full opacity-0 group-hover:opacity-100 transition-opacity duration-500 animate-pulse pointer-events-none" />
          </button>
          
          <div className="hidden md:flex gap-8 text-[12px] font-medium tracking-[0.15em] uppercase text-ice-white/70 pointer-events-auto">
            {[
              { label: 'Missions', id: 'services-section' },
              { label: 'Process', id: 'process-section' },
              { label: 'Telemetry', id: 'globe-section' },
              { label: 'Contact', id: 'contact-section' },
            ].map((item) => (
              <button 
                key={item.id}
                onClick={() => scrollTo(item.id)} 
                className="relative hover:text-electric-blue transition-colors duration-300 group/nav py-1"
              >
                {item.label}
                <span className="absolute -bottom-0.5 left-0 w-0 h-[1px] bg-electric-blue group-hover/nav:w-full transition-all duration-300" />
              </button>
            ))}
          </div>
        </div>
      </nav>

      {/* Content Sections */}
      <div className={!isLoaded ? 'opacity-0' : 'opacity-100 transition-opacity duration-1000 relative z-10'}>
        <Hero />
        
        <SectionDivider />
        <AnimatedStats />
        
        <SectionDivider />
        <Services />
        
        <SectionDivider />
        <BentoGrid />
        
        <SectionDivider />
        <LighthouseWidget />

        <Suspense fallback={<div style={{ height: '500vh', background: '#0a0a0f' }} />}>
          <Process />
        </Suspense>

        <SectionDivider />
        <TechStack />

        <Suspense fallback={<div style={{ height: '100vh', background: '#0a0a0f' }} />}>
          <GlobeSection />
        </Suspense>

        <SectionDivider />
        <Contact />
      </div>

      {/* Ambient Audio Toggle */}
      <AmbientAudio />

      {/* Premium Footer */}
      <footer className="relative z-10 gradient-border-top bg-void-black">
        <div className="max-w-7xl mx-auto py-16 px-6 md:px-12">
          {/* Footer top */}
          <div className="flex flex-col md:flex-row justify-between items-start md:items-center gap-8 mb-12 pb-12 border-b border-white/[0.04]">
            <div className="flex flex-col gap-3 group/brand">
              <div className="text-2xl font-bold tracking-[0.15em] uppercase font-display relative inline-block">
                 <span className="relative z-10 group-hover/brand:text-shadow-glow transition-all duration-500">TECHY</span>
                 <div className="absolute inset-0 bg-electric-blue/10 blur-lg opacity-0 group-hover/brand:opacity-100 transition-opacity duration-500 pointer-events-none" />
              </div>
              <p className="text-sm text-ice-white/40 max-w-xs leading-relaxed group-hover/brand:text-ice-white/60 transition-colors">

import React, { useState, useRef, useEffect } from 'react';
import { Volume2, VolumeX } from 'lucide-react';
import { cn } from '../lib/utils';
import { motion } from 'motion/react';

export function AmbientAudio() {
  const [isPlaying, setIsPlaying] = useState(false);
  const audioRef = useRef<HTMLAudioElement>(null);

  // Using a soft, deep space drone royalty-free ambient track
  const audioUrl = "https://cdn.pixabay.com/audio/2022/10/25/audio_249ea4dd41.mp3"; 

  useEffect(() => {
    if (audioRef.current) {
      audioRef.current.volume = 0.3; // keep it subtle
    }
  }, []);

  const togglePlay = () => {
    if (audioRef.current) {
      if (isPlaying) {
        audioRef.current.pause();
      } else {
        audioRef.current.play().catch(e => console.warn("Audio playback prevented by browser:", e));
      }
      setIsPlaying(!isPlaying);
    }
  };

  return (
    <div className="fixed bottom-6 right-6 z-50 flex items-center gap-3">
      {/* Equalizer bars (only animate when playing) */}
      <div className={cn("flex items-end gap-[2px] h-4 transition-opacity duration-500", isPlaying ? "opacity-100" : "opacity-0")}>
        {[...Array(4)].map((_, i) => (
          <motion.div
            key={i}
            className="w-1 bg-electric-blue rounded-t-sm"
            animate={isPlaying ? {
              height: ['20%', '100%', '40%', '80%', '20%']
            } : { height: '20%' }}
            transition={{
              duration: 1 + Math.random(),
              repeat: Infinity,
              ease: "easeInOut",
              delay: i * 0.1
            }}
          />
        ))}
      </div>

      <button
        onClick={togglePlay}
        className="w-10 h-10 rounded-full glass-pane border border-white/10 flex items-center justify-center text-white/50 hover:text-white hover:border-electric-blue/50 transition-all duration-300 group overflow-hidden relative"
        aria-label="Toggle ambient audio"
      >
        <div className="absolute inset-0 bg-electric-blue/10 opacity-0 group-hover:opacity-100 transition-opacity" />
        {isPlaying ? (
          <Volume2 className="w-5 h-5 relative z-10" />
        ) : (
          <VolumeX className="w-5 h-5 relative z-10" />
        )}
      </button>

      <audio 
        ref={audioRef}
        src={audioUrl}
        loop
        preload="auto"
      />
    </div>
  );
}

      setDisplay(isDecimal ? Math.round(val * 10) / 10 : Math.round(val));
      
      if (p < 1) {
        requestAnimationFrame(tick);
      } else if (!isFinished) {
        isFinished = true;
        setComplete(true);
        triggerBurst();
      }
    };
    requestAnimationFrame(tick);
  }, [active, target, triggerBurst]);

  return (
    <div ref={containerRef} className="relative inline-block">
      <span className={complete ? "text-holographic" : ""}>{display}</span>
      <span className="text-3xl ml-1 text-ice-white/60">{suffix}</span>
    </div>
  );
}

function CircularRing({ progress, color, size = 100 }: { progress: number; color: string; size?: number }) {
  const radius = (size - 12) / 2;
  const circumference = 2 * Math.PI * radius;
  const offset = circumference - (progress / 100) * circumference;

  return (
    <svg className="absolute inset-0 w-full h-full -rotate-90" viewBox={`0 0 ${size} ${size}`}>
      <circle cx={size/2} cy={size/2} r={radius} fill="none" stroke="rgba(255,255,255,0.03)" strokeWidth="3" />
      <circle
        cx={size/2} cy={size/2} r={radius}
        fill="none"
        stroke={color}
        strokeWidth="3"
        strokeLinecap="round"
        strokeDasharray={circumference}
        strokeDashoffset={offset}
        style={{
          transition: 'stroke-dashoffset 2.4s cubic-bezier(0.16, 1, 0.3, 1)',
          filter: `drop-shadow(0 0 6px ${color}60)`,
        }}
      />
    </svg>
  );
}

import React, { useEffect, useRef } from 'react';
import gsap from 'gsap';

export function CustomCursor() {
  const dotRef = useRef<HTMLDivElement>(null);
  const ringRef = useRef<HTMLDivElement>(null);
  const pos = useRef({ x: 0, y: 0 });
  const ringPos = useRef({ x: 0, y: 0 });

  useEffect(() => {
    const prefersReducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;
    if (prefersReducedMotion) return;

    const onMouseMove = (e: MouseEvent) => {
      pos.current = { x: e.clientX, y: e.clientY };
      
      // Dot follows instantly
      if (dotRef.current) {
        dotRef.current.style.left = `${e.clientX}px`;
        dotRef.current.style.top = `${e.clientY}px`;
      }
    };

    // Ring follows with lerp
    const updateRing = () => {
      ringPos.current.x += (pos.current.x - ringPos.current.x) * 0.15;
      ringPos.current.y += (pos.current.y - ringPos.current.y) * 0.15;
      
      if (ringRef.current) {
        ringRef.current.style.left = `${ringPos.current.x}px`;
        ringRef.current.style.top = `${ringPos.current.y}px`;
      }
      requestAnimationFrame(updateRing);
    };

    // Hover detection
    const onMouseOver = (e: MouseEvent) => {
      const target = e.target as HTMLElement;
      const isClickable = target.closest('a, button, [role="button"], input, select, textarea, [data-cursor="pointer"]');
      
      if (isClickable) {
        document.body.classList.add('cursor-hover');
      }
    };

    const onMouseOut = (e: MouseEvent) => {
      const target = e.target as HTMLElement;
      const isClickable = target.closest('a, button, [role="button"], input, select, textarea, [data-cursor="pointer"]');
      
      if (isClickable) {
        document.body.classList.remove('cursor-hover');
      }
    };

    const onMouseDown = () => document.body.classList.add('cursor-click');
    const onMouseUp = () => document.body.classList.remove('cursor-click');

    window.addEventListener('mousemove', onMouseMove);
    window.addEventListener('mouseover', onMouseOver);
    window.addEventListener('mouseout', onMouseOut);
    window.addEventListener('mousedown', onMouseDown);
    window.addEventListener('mouseup', onMouseUp);
    
    const frameId = requestAnimationFrame(updateRing);

    return () => {
      window.removeEventListener('mousemove', onMouseMove);
      window.removeEventListener('mouseover', onMouseOver);
      window.removeEventListener('mouseout', onMouseOut);
      window.removeEventListener('mousedown', onMouseDown);
      window.removeEventListener('mouseup', onMouseUp);
      cancelAnimationFrame(frameId);
    };
  }, []);

  return (
    <>
      <div ref={dotRef} className="cursor-dot" />
      <div ref={ringRef} className="cursor-ring" />
    </>
  );
}

function ConnectionLine({ start, end }: { start: [number, number, number], end: [number, number, number] }) {
  const lineRef = useRef<any>(null);

  useFrame(() => {
    if (lineRef.current) {
      lineRef.current.material.dashOffset -= 0.01;
    }
  });

  const points = [
    new THREE.Vector3(...start),
    new THREE.Vector3(0, 0, 0),
    new THREE.Vector3(...end)
  ];
  
  const curve = new THREE.QuadraticBezierCurve3(points[0], points[1], points[2]);
  const curvePoints = curve.getPoints(50);

  const [pointsPos] = React.useState(() => {
    const pos = new Float32Array(50 * 3);
    for(let i=0; i<50; i++) {
        const t = Math.random();
        const p = curve.getPoint(t);
        pos[i*3] = p.x;
        pos[i*3+1] = p.y;
        pos[i*3+2] = p.z;
    }
    return pos;
  });

  const particlesRef = useRef<any>(null);

  useFrame((state, delta) => {
    if (particlesRef.current) {
        const positions = particlesRef.current.geometry.attributes.position.array;
        for(let i=0; i<50; i++) {
            // Reconstruct t based on current position (approximate) or just move along curve
            // For simplicity, we just constantly update t and get new point
            const currentT = (particlesRef.current.userData[i] || Math.random() );
            let nextT = currentT + delta * 0.2;
            if (nextT > 1) nextT = 0;
            particlesRef.current.userData[i] = nextT;
            
            const p = curve.getPoint(nextT);
            positions[i*3] = p.x;
            positions[i*3+1] = p.y;
            positions[i*3+2] = p.z;
        }
        particlesRef.current.geometry.attributes.position.needsUpdate = true;
    }
  });


  return (
    <group>
      <Line
        ref={lineRef}
        points={curvePoints}
        color="#00D4FF"
        lineWidth={1.5}
        dashed={true}
        dashSize={0.2}
        dashScale={1}
        dashOffset={0}
        transparent
        opacity={0.3}
      />
      
      {/* Data stream particles along the line */}
      <Points ref={particlesRef} positions={pointsPos} stride={3}>
        <PointMaterial
          transparent
          color="#7c3aed"
          size={0.06}
          sizeAttenuation={true}
          depthWrite={false}
          blending={THREE.AdditiveBlending}
        />
      </Points>
    </group>
  );
}

import React, { useRef, useCallback } from 'react';
import { motion, useMotionValue, useSpring, useTransform } from 'motion/react';

interface MagneticButtonProps {
  children: React.ReactNode;
  className?: string;
  onClick?: () => void;
  type?: 'button' | 'submit';
  disabled?: boolean;
}

export function MagneticButton({ children, className = '', onClick, type = 'button', disabled = false }: MagneticButtonProps) {
  const ref = useRef<HTMLButtonElement>(null);
  const x = useMotionValue(0);
  const y = useMotionValue(0);

  const springX = useSpring(x, { stiffness: 200, damping: 20 });
  const springY = useSpring(y, { stiffness: 200, damping: 20 });

  const rotateX = useTransform(springY, [-0.5, 0.5], [8, -8]);
  const rotateY = useTransform(springX, [-0.5, 0.5], [-8, 8]);

  const handleMouseMove = useCallback((e: React.MouseEvent<HTMLButtonElement>) => {
    if (!ref.current) return;
    const rect = ref.current.getBoundingClientRect();
    const centerX = rect.left + rect.width / 2;
    const centerY = rect.top + rect.height / 2;
    const distX = e.clientX - centerX;
    const distY = e.clientY - centerY;
    const dist = Math.sqrt(distX * distX + distY * distY);
    
    // Only magnetic within 50px radius
    if (dist < 50) {
      x.set(distX * 0.3);
      y.set(distY * 0.3);
    }
  }, [x, y]);

  const handleMouseLeave = useCallback(() => {
    x.set(0);
    y.set(0);
  }, [x, y]);

  const handleClick = useCallback((e: React.MouseEvent<HTMLButtonElement>) => {
    // Ripple effect
    const btn = ref.current;
    if (!btn) return;
    
    const rect = btn.getBoundingClientRect();
    const ripple = document.createElement('span');
    const size = Math.max(rect.width, rect.height);
    ripple.className = 'ripple';
    ripple.style.width = ripple.style.height = `${size}px`;
    ripple.style.left = `${e.clientX - rect.left - size / 2}px`;
    ripple.style.top = `${e.clientY - rect.top - size / 2}px`;
    btn.appendChild(ripple);
    
    setTimeout(() => ripple.remove(), 600);
    
    onClick?.();
  }, [onClick]);

  return (
    <motion.button
      ref={ref}
      type={type}
      disabled={disabled}
      className={`relative group ripple-container glow-pulse bg-deep-space/40 backdrop-blur-sm border border-orbit-gray px-8 py-4 text-ice-white text-sm font-medium tracking-[0.1em] uppercase transition-all duration-300 hover:border-electric-blue hover:shadow-[0_0_40px_rgba(0,212,255,0.2)] overflow-hidden rounded-lg ${className}`}
      style={{
        x: springX,
        y: springY,
        rotateX,
        rotateY,
        transformStyle: 'preserve-3d',
        perspective: 1000,
      }}
      onMouseMove={handleMouseMove}
      onMouseLeave={handleMouseLeave}
      onClick={handleClick}
      whileTap={{ scale: 0.95 }}
    >
      {/* Corner accents */}
      <div className="absolute top-0 left-0 w-3 h-3 border-t border-l border-orbit-gray group-hover:border-electric-blue transition-colors duration-300" />

// â”€â”€ Canvas Particle Vortex + Energy Core â”€â”€
function useParticleVortex(
  canvasRef: React.RefObject<HTMLCanvasElement | null>,
  active: React.MutableRefObject<boolean>,
  phase: React.MutableRefObject<'vortex' | 'converge' | 'explode' | 'idle'>
) {
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d')!;

    let W = (canvas.width = window.innerWidth);
    let H = (canvas.height = window.innerHeight);
    let animId: number;

    const PARTICLE_COUNT = 2000;

    interface Particle {
      angle: number;
      radius: number;
      baseSpeed: number;
      radialSpeed: number;
      size: number;
      hue: number;
      alpha: number;
      trail: { x: number; y: number }[];
    }

    const particles: Particle[] = Array.from({ length: PARTICLE_COUNT }, () => {
      const maxDim = Math.max(W, H);
      return {
        angle: Math.random() * Math.PI * 2,
        radius: Math.random() * maxDim * 0.9 + 80,
        baseSpeed: (Math.random() * 0.025 + 0.006) * (Math.random() > 0.5 ? 1 : -1),
        radialSpeed: -(Math.random() * 1.8 + 0.4),
        size: Math.random() * 2.8 + 0.4,
        hue: Math.random() > 0.3 ? 195 : (Math.random() > 0.5 ? 270 : 210),
        alpha: Math.random() * 0.9 + 0.1,
        trail: [],
      };
    });

    let coreGlow = 0;

    const draw = () => {
      if (!active.current) { cancelAnimationFrame(animId); return; }

      ctx.fillStyle = 'rgba(0,0,0,0.1)';
      ctx.fillRect(0, 0, W, H);

      const centerX = W / 2;
      const centerY = H / 2;
      const currentPhase = phase.current;

      // Energy core
      const coreTarget = currentPhase === 'explode' ? 1.5
                       : currentPhase === 'converge' ? 0.8 : 0.15;
      coreGlow += (coreTarget - coreGlow) * 0.03;

      if (coreGlow > 0.05) {
        const r = 80 * coreGlow;
        const g = ctx.createRadialGradient(centerX, centerY, 0, centerX, centerY, r);
        g.addColorStop(0, `rgba(0,212,255,${coreGlow * 0.8})`);
        g.addColorStop(0.3, `rgba(124,58,237,${coreGlow * 0.4})`);
        g.addColorStop(1, 'transparent');
        ctx.beginPath();
        ctx.arc(centerX, centerY, r, 0, Math.PI * 2);
        ctx.fillStyle = g;
        ctx.fill();
      }

      // Particles
      for (const p of particles) {
        const speedMult = currentPhase === 'converge' ? 2.5 : 1;
        p.angle += p.baseSpeed * speedMult;

        if (currentPhase === 'explode') {
          p.radius += Math.abs(p.radialSpeed) * 6;
        } else {
          p.radius += p.radialSpeed * (currentPhase === 'converge' ? 1.8 : 1);
        }

        if (p.radius < 5 && currentPhase !== 'explode') {
          p.radius = Math.max(W, H) * (currentPhase === 'converge' ? 0.5 : 0.8) + Math.random() * 300;
          p.angle = Math.random() * Math.PI * 2;
        }
        if (p.radius > Math.max(W, H) * 1.5 && currentPhase !== 'explode') {
          p.radius = Math.max(W, H) * 0.8;
          p.angle = Math.random() * Math.PI * 2;
        }

        const x = centerX + Math.cos(p.angle) * p.radius;
        const y = centerY + Math.sin(p.angle) * p.radius;

        p.trail.push({ x, y });
        if (p.trail.length > 8) p.trail.shift();

        if (p.trail.length > 1) {
          ctx.beginPath();
          ctx.moveTo(p.trail[0].x, p.trail[0].y);
          for (let i = 1; i < p.trail.length; i++) ctx.lineTo(p.trail[i].x, p.trail[i].y);
          ctx.strokeStyle = `hsla(${p.hue}, 100%, 65%, ${p.alpha * 0.25})`;
          ctx.lineWidth = p.size * 0.4;
          ctx.stroke();
        }

        ctx.beginPath();
        ctx.arc(x, y, p.size, 0, Math.PI * 2);
        ctx.fillStyle = `hsla(${p.hue}, 100%, 72%, ${p.alpha})`;
        ctx.shadowBlur = p.size * 8;
        ctx.shadowColor = `hsla(${p.hue}, 100%, 60%, 0.9)`;
        ctx.fill();
        ctx.shadowBlur = 0;
      }

      animId = requestAnimationFrame(draw);
    };

    draw();

    const onResize = () => { W = canvas.width = window.innerWidth; H = canvas.height = window.innerHeight; };
    window.addEventListener('resize', onResize);
    return () => { active.current = false; cancelAnimationFrame(animId); window.removeEventListener('resize', onResize); };
  }, [canvasRef, active, phase]);
}

    vec4 mod289(vec4 x) { return x - floor(x * (1.0 / 289.0)) * 289.0; }

      vec3 l = 1.0 - g;

      vec3 i2 = max( g.xyz, l.zxy );
      vec3 x1 = x0 - i1 + C.xxx;
      vec3 x2 = x0 - i2 + C.yyy; // 2.0*C.x = 1/3 = C.y
      vec3 x3 = x0 - D.yyy;      // -1.0+3.0*C.x = -0.5 = -D.y
      i = mod289(i);
      vec4 p = permute( permute( permute(
                i.z + vec4(0.0, i1.z, i2.z, 1.0 ))
              + i.y + vec4(0.0, i1.y, i2.y, 1.0 ))
              + i.x + vec4(0.0, i1.x, i2.x, 1.0 ));

      vec4 x_ = floor(j * ns.z);
      vec4 y_ = floor(j - 7.0 * x_ );    // mod(j,N)
      vec4 x = x_ *ns.x + ns.yyyy;
      vec4 y = y_ *ns.x + ns.yyyy;
      vec4 h = 1.0 - abs(x) - abs(y);
      vec4 b0 = vec4( x.xy, y.xy );
      vec4 b1 = vec4( x.zw, y.zw );
      vec4 s0 = floor(b0)*2.0 + 1.0;
      vec4 s1 = floor(b1)*2.0 + 1.0;
      vec4 sh = -step(h, vec4(0.0));
      vec4 a0 = b0.xzyw + s0.xzyw*sh.xxyy ;
      vec4 a1 = b1.xzyw + s1.xzyw*sh.zzww ;
      vec3 p0 = vec3(a0.xy,h.x);
      vec3 p1 = vec3(a0.zw,h.y);
      vec3 p2 = vec3(a1.xy,h.z);
      vec3 p3 = vec3(a1.zw,h.w);
      vec4 norm = taylorInvSqrt(vec4(dot(p0,p0), dot(p1,p1), dot(p2, p2), dot(p3,p3)));
      p0 *= norm.x;
      p1 *= norm.y;
      p2 *= norm.z;
      p3 *= norm.w;
      vec4 m = max(0.6 - vec4(dot(x0,x0), dot(x1,x1), dot(x2,x2), dot(x3,x3)), 0.0);
      m = m * m;
      return 42.0 * dot( m*m, vec4( dot(p0,x0), dot(p1,x1),
                                    dot(p2,x2), dot(p3,x3) ) );
    }
    
    void main() {

function NeutronScene({ scrollProgress, isMobile }: { scrollProgress: React.MutableRefObject<number>, isMobile: boolean }) {
  const groupRef = useRef<THREE.Group>(null);
  const particlesRef = useRef<THREE.Points>(null);
  const shockwaveRef = useRef<THREE.Mesh>(null);
  const flashRef = useRef<THREE.Mesh>(null);
  const starsRef = useRef<THREE.Group>(null);
  
  const hasExploded = useRef(false);
  const explosionTime = useRef(0);

  // Create a soft glowing texture for particles to make them look realistic
  const particleTexture = useMemo(() => {
    const canvas = document.createElement('canvas');
    canvas.width = 64;
    canvas.height = 64;
    const context = canvas.getContext('2d')!;
    const gradient = context.createRadialGradient(32, 32, 0, 32, 32, 32);
    gradient.addColorStop(0, 'rgba(255, 255, 255, 1)');
    gradient.addColorStop(0.2, 'rgba(255, 255, 255, 0.8)');
    gradient.addColorStop(0.6, 'rgba(255, 255, 255, 0.2)');
    gradient.addColorStop(1, 'rgba(0, 0, 0, 0)');
    context.fillStyle = gradient;
    context.fillRect(0, 0, 64, 64);
    return new THREE.CanvasTexture(canvas);
  }, []);
  
  useFrame((state, delta) => {
    const time = state.clock.getElapsedTime();
    const progress = scrollProgress.current; // 0 to 5

    // Global scene rotation based on scroll for "moving through space" feel
    if (starsRef.current) {
        starsRef.current.rotation.y = progress * 0.1;
        starsRef.current.rotation.x = progress * 0.05;
        starsRef.current.position.z = -progress * 0.5; // Move forward slightly
    }

    // Optimization: Stop heavy particle calculations if we are far past the explosion
    // and it has fully faded out.
    if (hasExploded.current && explosionTime.current > 15) {
        return;
    }

    if (groupRef.current) {
      // Phase 0-3: Neutron Stars logic
      
      // Rotation: Spin faster as we approach explosion (Phase 3)
      const spinSpeed = progress < 3 ? 0.5 + progress * 2 : 0;
      groupRef.current.rotation.z += delta * spinSpeed;

      // Scale: Spiral in before explosion (between Phase 2 and 3)
      let scale = 1;
      if (progress > 2 && progress < 3) {
          scale = THREE.MathUtils.lerp(1, 0.1, (progress - 2));
      } else if (progress >= 3) {
          scale = 0; // Hide
      }
      groupRef.current.scale.set(scale, scale, scale);
      
      // Position: Move from side to center
      const initialX = isMobile ? 0 : 2;
      const initialY = isMobile ? 1.5 : 0;
      
      // Smoothly interpolate position based on progress
      // 0 -> initial
      // 2.5 -> center (0,0) - Slower approach
      const moveFactor = Math.min(progress / 2.5, 1); 
      // Use easeOut for smoother arrival
      const ease = 1 - Math.pow(1 - moveFactor, 3);
      
      groupRef.current.position.x = THREE.MathUtils.lerp(initialX, 0, ease);
      groupRef.current.position.y = THREE.MathUtils.lerp(initialY, 0, ease);
    }

    if (particlesRef.current) {
        if (progress >= 3 && !hasExploded.current) {
            hasExploded.current = true;
            explosionTime.current = 0;
        } else if (progress < 2.5 && hasExploded.current) {
            hasExploded.current = false;
            explosionTime.current = 0;
        }

        if (hasExploded.current) {
            explosionTime.current += delta;
            particlesRef.current.visible = true;
            
            const positions = particlesRef.current.geometry.attributes.position.array as Float32Array;
            const velocities = particlesRef.current.userData.velocities as Float32Array;
            
            // Frame-rate independent drag (roughly equivalent to 0.992 at 60fps)
            // Less drag means particles travel further and spread out more
            const drag = Math.pow(0.992, delta * 60);
            
            for(let i = 0; i < positions.length; i+=3) {
                // Add a slight outward drift that persists even when velocity drops
                // Increased drift for more spread
                const driftSpeed = 0.15 * delta;
                const dist = Math.sqrt(positions[i]*positions[i] + positions[i+1]*positions[i+1] + positions[i+2]*positions[i+2]) || 1;
                const dirX = positions[i] / dist;
                const dirY = positions[i+1] / dist;
                const dirZ = positions[i+2] / dist;

                // Slower initial burst (1.2 instead of 2.0), but travels further due to lower drag
                positions[i] += velocities[i] * delta * 1.2 + dirX * driftSpeed;
                positions[i+1] += velocities[i+1] * delta * 1.2 + dirY * driftSpeed;
                positions[i+2] += velocities[i+2] * delta * 1.2 + dirZ * driftSpeed;
                
                velocities[i] *= drag;
                velocities[i+1] *= drag;
                velocities[i+2] *= drag;
            }
            particlesRef.current.geometry.attributes.position.needsUpdate = true;
            
            // Smoother exponential fade out, lasts even longer (~15 seconds)
            const targetOpacity = Math.max(0, 1.0 - Math.pow(explosionTime.current / 15.0, 1.5));
            (particlesRef.current.material as THREE.PointsMaterial).opacity = targetOpacity;

        } else {
            particlesRef.current.visible = false;
            
            const positions = particlesRef.current.geometry.attributes.position.array as Float32Array;
            const originalPositions = particlesRef.current.userData.originalPositions as Float32Array;
            const velocities = particlesRef.current.userData.velocities as Float32Array;
            const originalVelocities = particlesRef.current.userData.originalVelocities as Float32Array;
            
            for(let i = 0; i < positions.length; i++) {
                positions[i] = originalPositions[i];
                velocities[i] = originalVelocities[i];
            }
            particlesRef.current.geometry.attributes.position.needsUpdate = true;
            (particlesRef.current.material as THREE.PointsMaterial).opacity = 1;
        }
    }

    // Supernova Blinding Flash
    if (flashRef.current) {
        if (hasExploded.current) {
            flashRef.current.visible = true;
            // Smooth ease-out expansion, slightly slower
            const flashScale = 1 + 30 * (1 - Math.exp(-explosionTime.current * 2.5));
            flashRef.current.scale.set(flashScale, flashScale, flashScale);
            // Smooth fade out, lingers longer
            (flashRef.current.material as THREE.MeshBasicMaterial).opacity = Math.max(0, 1.0 - explosionTime.current * 0.8);
        } else {
            flashRef.current.visible = false;
            flashRef.current.scale.set(0.1, 0.1, 0.1);
            (flashRef.current.material as THREE.MeshBasicMaterial).opacity = 1;
        }
    }

    // Shockwave ring - smoother expansion
    if (shockwaveRef.current) {
        if (hasExploded.current) {
            shockwaveRef.current.visible = true;
            // Smooth ease-out expansion, slightly slower
            const scale = 0.1 + 30 * (1 - Math.exp(-explosionTime.current * 1.2));
            shockwaveRef.current.scale.set(scale, scale, scale);
            // Smooth fade out, lingers longer
            (shockwaveRef.current.material as THREE.MeshBasicMaterial).opacity = Math.max(0, 0.8 - explosionTime.current * 0.1);
        } else {
            shockwaveRef.current.visible = false;
            shockwaveRef.current.scale.set(0.1, 0.1, 0.1);
            (shockwaveRef.current.material as THREE.MeshBasicMaterial).opacity = 0.8;
        }
    }
  });

  // Particle explosion data - Increased count for massive supernova
  const particleCount = 8000;
  const [positions, colors, velocities, originalVelocities] = useMemo(() => {
    const pos = new Float32Array(particleCount * 3);
    const cols = new Float32Array(particleCount * 3);
    const vels = new Float32Array(particleCount * 3);
    
    const colorPalette = [
      new THREE.Color('#FFD700'), // Yellow
      new THREE.Color('#FF4500'), // Red
      new THREE.Color('#00D4FF'), // Blue
      new THREE.Color('#8A2BE2'), // Purple
      new THREE.Color('#FFFFFF')  // White Core
    ];
    
    for (let i = 0; i < particleCount; i++) {
      // Sphere distribution
      const r = Math.random();
      const theta = Math.random() * Math.PI * 2;
      const phi = Math.acos(2 * Math.random() - 1);
      
      const x = r * Math.sin(phi) * Math.cos(theta);
      const y = r * Math.sin(phi) * Math.sin(theta);
      const z = r * Math.cos(phi);
      
      pos[i * 3] = x * 0.1; // Start small
      pos[i * 3 + 1] = y * 0.1;
      pos[i * 3 + 2] = z * 0.1;
      
      // Explosion velocity
      const speed = Math.random() * 1.5 + 0.2;
      vels[i * 3] = x * speed;
      vels[i * 3 + 1] = y * speed;
      vels[i * 3 + 2] = z * speed;
      
      // Color mixing
      const color = colorPalette[Math.floor(Math.random() * colorPalette.length)];
      
      cols[i * 3] = color.r;
      cols[i * 3 + 1] = color.g;
      cols[i * 3 + 2] = color.b;
    }
    return [pos, cols, vels, new Float32Array(vels)];
  }, []);

  return (
    <>
      {/* Integrated StarField for unified motion */}
      <group ref={starsRef}>
         <StarMesh />
         <ShootingStar />
      </group>

      <group ref={groupRef} position={[2, 0, 0]}>
        <NeutronStar position={[0.6, 0, 0]} color="#00D4FF" intensity={2.5} />
        <NeutronStar position={[-0.6, 0, 0]} color="#FF4500" intensity={3.0} />
      </group>

      {/* Supernova Flash */}
      <mesh ref={flashRef} visible={false}>
        <sphereGeometry args={[0.5, 32, 32]} />
        <meshBasicMaterial color="#FFFFFF" transparent opacity={1} blending={THREE.AdditiveBlending} depthWrite={false} />
      </mesh>

      <Points 
        ref={particlesRef} 
        positions={positions} 
        colors={colors} 
        stride={3} 
        visible={false}
        userData={{ velocities, originalPositions: positions.slice(), originalVelocities }}
      >
        <PointMaterial
            transparent
            vertexColors
            size={0.15}
            sizeAttenuation={true}
            depthWrite={false}
            blending={THREE.AdditiveBlending}
            map={particleTexture}
            alphaTest={0.001}

      {/* Shockwave Ring */}
      <mesh ref={shockwaveRef} rotation={[Math.PI / 2, 0, 0]} visible={false}>
        <ringGeometry args={[0.8, 1, 64]} />
        <meshBasicMaterial color="#00D4FF" transparent opacity={0.8} side={THREE.DoubleSide} blending={THREE.AdditiveBlending} />
      </mesh>

    </>
  );

        const dist = Math.sqrt(Math.pow(x - mouse.x * viewport.width / 2, 2) + Math.pow(y - mouse.y * viewport.height / 2, 2));
        const force = Math.max(0, 1 - dist / 3);
        
        posAttr.array[idx + 2] = Math.sin(time + x + y) * 0.2 + force * 0.5;
      }
    }
    posAttr.needsUpdate = true;
  });

  return (
    <points ref={meshRef}>
      <bufferGeometry>
        <bufferAttribute
          attach="attributes-position"
          count={positions.length / 3}
          array={positions}
          itemSize={3}
        />
        <bufferAttribute
          attach="attributes-color"
          count={colors.length / 3}
          array={colors}
          itemSize={3}
        />
      </bufferGeometry>
      <pointsMaterial size={0.05} vertexColors transparent opacity={0.4} />
    </points>
  );
}

function Jellyfish() {

  const groupRef = useRef<THREE.Group>(null);
  const { scrollYProgress } = useScroll();
  
  // Scroll-based transforms - peaking at 0.5
  const scale = useTransform(scrollYProgress, [0, 0.4, 0.5, 0.6], [1, 1.5, 12, 0]);
  const distortion = useTransform(scrollYProgress, [0, 0.4, 0.5, 0.6], [0.4, 0.8, 4, 0]);
  const opacity = useTransform(scrollYProgress, [0, 0.45, 0.5, 0.6], [1, 1, 0, 0]);
  
  const springScale = useSpring(scale, { damping: 20, stiffness: 100 });

  useFrame((state) => {
    if (!groupRef.current || !headRef.current) return;
    const time = state.clock.getElapsedTime();
    
    // Pulsing motion
    const pulse = Math.sin(time * 2) * 0.1 + 1;
    groupRef.current.scale.setScalar(springScale.get() * pulse);
    
    // Gentle rotation
    groupRef.current.rotation.y = time * 0.2;
    groupRef.current.rotation.z = Math.sin(time * 0.5) * 0.1;
    
    // Head distortion pulse
    if (headRef.current.material instanceof THREE.MeshPhysicalMaterial || (headRef.current.material as any).distort !== undefined) {
      (headRef.current.material as any).distort = distortion.get() + Math.sin(time * 4) * 0.05;
    }
  });

  const tentacles = useMemo(() => {
    return Array.from({ length: 8 }).map((_, i) => {
      const angle = (i / 8) * Math.PI * 2;
      return {
        position: [Math.cos(angle) * 0.5, -0.5, Math.sin(angle) * 0.5] as [number, number, number],
        delay: i * 0.2
      };
    });
  }, []);

  return (
    <group ref={groupRef}>
      {/* Jellyfish Head */}
      <Sphere ref={headRef} args={[1, 64, 64]}>
        <MeshDistortMaterial
          color="#4F46E5"
          speed={2}

      cols[i * 3 + 2] = color.b;

        transparent
        vertexColors
        size={0.12}
        sizeAttenuation
        depthWrite={false}

function TerminalTyping({ text, delay = 0 }: { text: string; delay?: number }) {

    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting && !isTyping && displayedText === '') {
          setIsTyping(true);
          setTimeout(() => {
            let i = 0;
            const intervalId = setInterval(() => {
              setDisplayedText(text.slice(0, i + 1));
              i++;
              if (i === text.length) {
                clearInterval(intervalId);
                setIsTyping(false);
              }
            }, 30);
          }, delay);
        }
      },
      { threshold: 0.5 }
    );

    if (containerRef.current) {
      observer.observe(containerRef.current);
    }

    return () => observer.disconnect();
  }, [text, delay, isTyping, displayedText]);

  return (
    <div ref={containerRef} className="text-base text-ice-white/40 leading-relaxed font-mono">
      <span className="text-electric-blue mr-2">&gt;</span>
      {displayedText}
      <span className="inline-block w-2 h-4 bg-electric-blue ml-1 animate-pulse" style={{ opacity: isTyping ? 1 : '' }} />

export function StarMesh({ isInteractive = false }: { isInteractive?: boolean }) {
  const pointsRef = useRef<THREE.Points>(null);
  const groupRef = useRef<THREE.Group>(null);
  
  const count = 6000; // Increased count for better density
  const [positions, colors] = useMemo(() => {
    const pos = new Float32Array(count * 3);
    const cols = new Float32Array(count * 3);
    const color = new THREE.Color('#FFFFFF');
    
    for (let i = 0; i < count; i++) {
      const x = (Math.random() - 0.5) * 120;
      const y = (Math.random() - 0.5) * 120;
      const z = (Math.random() - 0.5) * 80;
      pos[i * 3] = x;
      pos[i * 3 + 1] = y;
      pos[i * 3 + 2] = z;
      
      // Random opacity/brightness simulation via color
      // Mix of white and faint blue
      const isBlue = Math.random() > 0.8;
      const baseColor = isBlue ? new THREE.Color('#00D4FF') : color;
      const brightness = Math.random() * 0.8 + 0.2;
      
      cols[i * 3] = baseColor.r * brightness;
      cols[i * 3 + 1] = baseColor.g * brightness;
      cols[i * 3 + 2] = baseColor.b * brightness;
    }
    return [pos, cols];
  }, []);

  useFrame((state, delta) => {
    // Continuous rotation
    if (pointsRef.current) {
      pointsRef.current.rotation.x -= delta / 50;
      pointsRef.current.rotation.y -= delta / 60;
    }

    // Mouse response (Parallax/Tilt)

       const { x, y } = state.pointer;
       // Smoothly lerp to target rotation based on mouse position
       // Increased sensitivity for a noticeable effect

       groupRef.current.rotation.y = THREE.MathUtils.lerp(groupRef.current.rotation.y, x * 0.15, 0.05);
    }
  });

  return (
    <group ref={groupRef} rotation={[0, 0, Math.PI / 4]}>
      <Points ref={pointsRef} positions={positions} colors={colors} stride={3} frustumCulled={false}>

export function ShootingStar() {
  const ref = useRef<any>(null);
  const startTime = useRef(0);
  const startPos = useRef(new THREE.Vector3());
  const endPos = useRef(new THREE.Vector3());
  const active = useRef(false);

  useFrame((state) => {
    if (!ref.current) return;
    const time = state.clock.getElapsedTime();

    if (!active.current) {
      if (time - startTime.current > 3) { // Every 3 seconds
        active.current = true;
        startTime.current = time;
        
        // Random start position
        startPos.current.set(
          (Math.random() - 0.5) * 40,
          (Math.random() - 0.5) * 40,
          (Math.random() - 0.5) * 20
        );
        
        // Random direction
        endPos.current.copy(startPos.current).add(
          new THREE.Vector3(
            (Math.random() - 0.5) * 20,
            (Math.random() - 0.5) * 20,
            (Math.random() - 0.5) * 10
          )
        );
      }
    } else {
      const progress = (time - startTime.current) * 2; // Speed
      if (progress > 1) {
        active.current = false;
        ref.current.visible = false;
      } else {
        ref.current.visible = true;
        const currentPos = new THREE.Vector3().lerpVectors(startPos.current, endPos.current, progress);
        const tailPos = new THREE.Vector3().lerpVectors(startPos.current, endPos.current, Math.max(0, progress - 0.1));
        
        const positions = new Float32Array([
          currentPos.x, currentPos.y, currentPos.z,
          tailPos.x, tailPos.y, tailPos.z
        ]);
        
        ref.current.geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
        ref.current.geometry.attributes.position.needsUpdate = true;
        ref.current.material.opacity = 1 - progress;
      }
    }
  });

  return (
    <line ref={ref}>
      <bufferGeometry />
      <lineBasicMaterial color="#00D4FF" transparent opacity={0} linewidth={2} />
    </line>
  );
}

import React, { useEffect, useRef, useState, useCallback } from 'react';
import { MousePointer2, Info, ArrowRight } from 'lucide-react';

// --- Types ---

interface Particle {
  x: number;
  y: number;
  originX: number;
  originY: number;
  vx: number;
  vy: number;
  size: number;
  color: string;
  angle: number; // For some organic oscillation
}

interface BackgroundParticle {
  x: number;
  y: number;
  vx: number;
  vy: number;
  size: number;
  alpha: number;
  phase: number;
}

interface MouseState {
  x: number;
  y: number;
  isActive: boolean;
}

// --- Configuration Constants ---

const PARTICLE_DENSITY = 0.00015; // Particles per pixel squared (adjust for density)
const BG_PARTICLE_DENSITY = 0.00005; // Less dense for background
const MOUSE_RADIUS = 180; // Radius of mouse influence
const RETURN_SPEED = 0.08; // How fast particles fly back to origin (spring constant)
const DAMPING = 0.90; // Friction (velocity decay)
const REPULSION_STRENGTH = 1.2; // Multiplier for mouse push force

// --- Helper Functions ---

const randomRange = (min: number, max: number) => Math.random() * (max - min) + min;

// --- Components ---

export const AntiGravityCanvas: React.FC = () => {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const containerRef = useRef<HTMLDivElement>(null);
  const [debugInfo, setDebugInfo] = useState({ count: 0, fps: 0 });
  
  // Mutable state refs to avoid re-renders during animation loop
  const particlesRef = useRef<Particle[]>([]);
  const backgroundParticlesRef = useRef<BackgroundParticle[]>([]);
  const mouseRef = useRef<MouseState>({ x: -1000, y: -1000, isActive: false });
  const frameIdRef = useRef<number>(0);
  const lastTimeRef = useRef<number>(0);

  // Initialize Particles
  const initParticles = useCallback((width: number, height: number) => {
    // 1. Main Interactive Particles
    const particleCount = Math.floor(width * height * PARTICLE_DENSITY);
    const newParticles: Particle[] = [];
    
    for (let i = 0; i < particleCount; i++) {
      const x = Math.random() * width;
      const y = Math.random() * height;
      
      newParticles.push({
        x: x,
        y: y,
        originX: x,
        originY: y,
        vx: 0,
        vy: 0,
        size: randomRange(1, 2.5), 
        color: Math.random() > 0.9 ? '#4285F4' : '#ffffff', 
        angle: Math.random() * Math.PI * 2,
      });
    }
    particlesRef.current = newParticles;

    // 2. Background Ambient Particles (Stars/Dust)
    const bgCount = Math.floor(width * height * BG_PARTICLE_DENSITY);
    const newBgParticles: BackgroundParticle[] = [];
    
    for (let i = 0; i < bgCount; i++) {
      newBgParticles.push({
        x: Math.random() * width,
        y: Math.random() * height,
        vx: (Math.random() - 0.5) * 0.2, // Very slow drift
        vy: (Math.random() - 0.5) * 0.2,
        size: randomRange(0.5, 1.5),
        alpha: randomRange(0.1, 0.4),
        phase: Math.random() * Math.PI * 2 // For twinkling offset
      });
    }
    backgroundParticlesRef.current = newBgParticles;

    setDebugInfo(prev => ({ ...prev, count: particleCount + bgCount }));
  }, []);

  // Animation Loop
  const animate = useCallback((time: number) => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    // Calculate Delta Time for smooth animation (optional, but good for FPS calculation)
    const delta = time - lastTimeRef.current;
    lastTimeRef.current = time;
    if (delta > 0) {
        setDebugInfo(prev => ({ ...prev, fps: Math.round(1000 / delta) }));
    }

    // Clear Canvas
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // --- Background Effects ---
    
    // 1. Pulsating Radial Glow
    const centerX = canvas.width / 2;
    const centerY = canvas.height / 2;
    const pulseSpeed = 0.0008;
    // Oscillates between 0.05 and 0.12 opacity
    const pulseOpacity = Math.sin(time * pulseSpeed) * 0.035 + 0.085; 
    
    const gradient = ctx.createRadialGradient(
        centerX, centerY, 0, 
        centerX, centerY, Math.max(canvas.width, canvas.height) * 0.7
    );
    gradient.addColorStop(0, `rgba(66, 133, 244, ${pulseOpacity})`); // Faint Google Blue
    gradient.addColorStop(1, 'rgba(0, 0, 0, 0)');
    
    ctx.fillStyle = gradient;
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    // 2. Background Particles (Drifting Stars)
    const bgParticles = backgroundParticlesRef.current;
    ctx.fillStyle = "#ffffff";
    
    for (let i = 0; i < bgParticles.length; i++) {
      const p = bgParticles[i];
      p.x += p.vx;
      p.y += p.vy;
      
      // Wrap around screen
      if (p.x < 0) p.x = canvas.width;
      if (p.x > canvas.width) p.x = 0;
      if (p.y < 0) p.y = canvas.height;
      if (p.y > canvas.height) p.y = 0;

      // Twinkle effect
      const twinkle = Math.sin(time * 0.002 + p.phase) * 0.5 + 0.5; // 0 to 1
      const currentAlpha = p.alpha * (0.3 + 0.7 * twinkle);

      ctx.globalAlpha = currentAlpha;
      ctx.beginPath();
      ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
      ctx.fill();
    }
    ctx.globalAlpha = 1.0; // Reset alpha for foreground

    // --- Main Foreground Physics ---

    const particles = particlesRef.current;
    const mouse = mouseRef.current;

    // Phase 1: Apply Forces (Mouse & Spring)
    for (let i = 0; i < particles.length; i++) {
      const p = particles[i];

      // 1. Calculate Distance to Mouse
      const dx = mouse.x - p.x;
      const dy = mouse.y - p.y;
      const distance = Math.sqrt(dx * dx + dy * dy);
      
      // 2. Mouse Repulsion Force
      if (mouse.isActive && distance < MOUSE_RADIUS) {
        const forceDirectionX = dx / distance;
        const forceDirectionY = dy / distance;
        const force = (MOUSE_RADIUS - distance) / MOUSE_RADIUS; 
        
        const repulsion = force * REPULSION_STRENGTH;
        p.vx -= forceDirectionX * repulsion * 5; 
        p.vy -= forceDirectionY * repulsion * 5;
      }

      // 3. Spring Force (Return to Origin)
      const springDx = p.originX - p.x;
      const springDy = p.originY - p.y;
      
      p.vx += springDx * RETURN_SPEED;
      p.vy += springDy * RETURN_SPEED;
    }

    // Phase 2: Resolve Collisions
    for (let i = 0; i < particles.length; i++) {
      for (let j = i + 1; j < particles.length; j++) {
        const p1 = particles[i];
        const p2 = particles[j];

        const dx = p2.x - p1.x;
        const dy = p2.y - p1.y;
        const distSq = dx * dx + dy * dy;
        const minDist = p1.size + p2.size;

        if (distSq < minDist * minDist) {
          const dist = Math.sqrt(distSq);
          
          if (dist > 0.01) { // Avoid division by zero
            const nx = dx / dist; // Normal X
            const ny = dy / dist; // Normal Y

            // Static Resolution: Push particles apart so they don't overlap
            const overlap = minDist - dist;
            const pushX = nx * overlap * 0.5;
            const pushY = ny * overlap * 0.5;

            p1.x -= pushX;
            p1.y -= pushY;
            p2.x += pushX;
            p2.y += pushY;

            // Dynamic Resolution: Elastic Collision
            // Relative velocity
            const dvx = p1.vx - p2.vx;
            const dvy = p1.vy - p2.vy;

            // Calculate velocity along the normal
            // Dot product of velocity difference and normal direction
            const velocityAlongNormal = dvx * nx + dvy * ny;

            // Only bounce if they are moving towards each other
            if (velocityAlongNormal > 0) {
              const m1 = p1.size; // Use size as mass proxy
              const m2 = p2.size;
              const restitution = 0.85; // Bounciness (1 is perfectly elastic)

              // Impulse scalar
              const impulseMagnitude = (-(1 + restitution) * velocityAlongNormal) / (1/m1 + 1/m2);

              // Apply impulse
              const impulseX = impulseMagnitude * nx;
              const impulseY = impulseMagnitude * ny;

              p1.vx += impulseX / m1;
              p1.vy += impulseY / m1;
              p2.vx -= impulseX / m2;
              p2.vy -= impulseY / m2;
            }
          }
        }
      }
    }

    // Phase 3: Integration & Drawing
    for (let i = 0; i < particles.length; i++) {
      const p = particles[i];

      // Physics Update
      p.vx *= DAMPING;
      p.vy *= DAMPING;

      p.x += p.vx;
      p.y += p.vy;

      // Drawing
      ctx.beginPath();
      ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
      
      const velocity = Math.sqrt(p.vx * p.vx + p.vy * p.vy);
      const opacity = Math.min(0.3 + velocity * 0.1, 1); 
      
      ctx.fillStyle = p.color === '#ffffff' 
        ? `rgba(255, 255, 255, ${opacity})` 
        : p.color;
      
      ctx.fill();
    }

    frameIdRef.current = requestAnimationFrame(animate);
  }, []);

  // Resize Handler
  useEffect(() => {
    const handleResize = () => {
      if (containerRef.current && canvasRef.current) {
        const { width, height } = containerRef.current.getBoundingClientRect();
        const dpr = window.devicePixelRatio || 1;
        
        // Set actual size in memory (scaled to account for extra pixel density)
        canvasRef.current.width = width * dpr;
        canvasRef.current.height = height * dpr;
        
        // Make it visible size
        canvasRef.current.style.width = `${width}px`;
        canvasRef.current.style.height = `${height}px`;

        // Normalize coordinate system to use CSS pixels
        const ctx = canvasRef.current.getContext('2d');
        if (ctx) ctx.scale(dpr, dpr);

        // Re-init particles for new dimensions
        initParticles(width, height);
      }
    };

    window.addEventListener('resize', handleResize);
    handleResize(); // Initial call

    return () => window.removeEventListener('resize', handleResize);
  }, [initParticles]);

  // Start Animation
  useEffect(() => {
    frameIdRef.current = requestAnimationFrame(animate);
    return () => cancelAnimationFrame(frameIdRef.current);
  }, [animate]);

  // Global Mouse Handlers to bypass z-index blocks
  useEffect(() => {
    const handleGlobalMouseMove = (e: MouseEvent) => {
      if (!containerRef.current) return;
      const rect = containerRef.current.getBoundingClientRect();
      const x = e.clientX - rect.left;
      const y = e.clientY - rect.top;
      
      // Check if mouse is roughly inside the hero section bounds
      if (y >= -100 && y <= rect.height + 100) {
        mouseRef.current = { x, y, isActive: true };
      } else {
        mouseRef.current.isActive = false;
      }
    };

    const handleGlobalMouseLeave = () => {
      mouseRef.current.isActive = false;
    };

    window.addEventListener('mousemove', handleGlobalMouseMove);
    document.addEventListener('mouseleave', handleGlobalMouseLeave);

    return () => {
      window.removeEventListener('mousemove', handleGlobalMouseMove);
      document.removeEventListener('mouseleave', handleGlobalMouseLeave);
    };
  }, []);

  return (
    <div 
      ref={containerRef} 
      className="absolute inset-0 z-0 overflow-hidden bg-transparent pointer-events-none"
    >
      <canvas ref={canvasRef} className="block w-full h-full" />
      
      {/* Debug Info Overlay (Hidden in production usually, but cool for tech demos) */}
      <div className="absolute bottom-4 right-4 pointer-events-none text-xs text-white/20 font-mono text-right">
        <p>{debugInfo.count} entities</p>
        <p>{debugInfo.fps} FPS</p>

};
